CREATE TABLE IF NOT EXISTS jobs_users (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    full_name VARCHAR(150) NOT NULL,
    email VARCHAR(190) NULL,
    phone VARCHAR(25) NULL,
    password_hash VARCHAR(255) NOT NULL,
    status ENUM('active', 'inactive', 'blocked') NOT NULL DEFAULT 'active',
    email_verified_at DATETIME NULL,
    last_login_at DATETIME NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    UNIQUE KEY uq_users_email (email),
    UNIQUE KEY uq_users_phone (phone),
    KEY idx_users_status (status)
) ;

        'higher_secondary',
        'vocational',
        'diploma',
        'graduate',
        'postgraduate'
    ) NULL,

        'none',
        'under_5000',
        '5000_20000',
        '20000_50000',
        '50000_100000',
        'above_100000'
    ) NULL,

        'jobs_only',

    questionnaire_completed_level ENUM('quick', 'detailed') NOT NULL DEFAULT 'quick',

CREATE TABLE IF NOT EXISTS jobs_opportunity_categories (
    id INT UNSIGNED NOT NULL AUTO_INCREMENT,
    name VARCHAR(120) NOT NULL,
    slug VARCHAR(140) NOT NULL,
    description TEXT NULL,
    sort_order INT NOT NULL DEFAULT 0,
    status ENUM('active', 'inactive') NOT NULL DEFAULT 'active',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    UNIQUE KEY uq_opportunity_categories_slug (slug),
    UNIQUE KEY uq_opportunity_categories_name (name),
    KEY idx_opportunity_categories_status (status)
) ;

CREATE TABLE IF NOT EXISTS jobs_opportunities (
    id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    category_id INT UNSIGNED NOT NULL,
    title VARCHAR(255) NOT NULL,
    slug VARCHAR(280) NOT NULL,
    short_summary VARCHAR(500) NOT NULL,
    full_description LONGTEXT NULL,

    opportunity_type ENUM(
        'job',
        'home_based_work',
        'self_employment',
        'micro_business'
    ) NOT NULL,

    suitable_for_text TEXT NULL,
    not_suitable_for_text TEXT NULL,

    min_age TINYINT UNSIGNED NULL,
    max_age TINYINT UNSIGNED NULL,

    min_education_level ENUM(
        'none',
        'primary',
        'middle',
        'secondary',
        'higher_secondary',
        'vocational',
        'diploma',
        'graduate',
        'postgraduate'
    ) NOT NULL DEFAULT 'none',

    preferred_education_level ENUM(
        'none',
        'primary',
        'middle',
        'secondary',
        'higher_secondary',
        'vocational',
        'diploma',
        'graduate',
        'postgraduate'
    ) NULL,

    prior_experience_required ENUM('no', 'preferred', 'required') NOT NULL DEFAULT 'no',
    digital_literacy_level ENUM('none', 'basic', 'moderate', 'strong') NOT NULL DEFAULT 'none',
    manual_effort_level ENUM('low', 'medium', 'high') NOT NULL DEFAULT 'low',
    home_based_suitability ENUM('yes', 'partial', 'no') NOT NULL DEFAULT 'no',

    investment_min DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    investment_max DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    earning_min DECIMAL(12,2) NULL,
    earning_max DECIMAL(12,2) NULL,

    time_to_start_label ENUM(
        'immediate',
        'within_1_week',
        'within_1_month',
        'more_than_1_month'
    ) NOT NULL DEFAULT 'within_1_month',

    growth_potential ENUM('low', 'medium', 'high') NOT NULL DEFAULT 'medium',

    land_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    shop_space_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    internet_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    computer_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    smartphone_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    vehicle_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    tools_required ENUM('no', 'helpful', 'required') NOT NULL DEFAULT 'no',
    tools_required_text VARCHAR(500) NULL,

    family_support_helpful ENUM('no', 'yes') NOT NULL DEFAULT 'no',
    physical_effort_level ENUM('low', 'medium', 'high') NOT NULL DEFAULT 'low',
    risk_level ENUM('low', 'medium', 'high') NOT NULL DEFAULT 'medium',
    scalability_level ENUM('low', 'medium', 'high') NOT NULL DEFAULT 'medium',

    urban_suitability ENUM('high', 'medium', 'low') NOT NULL DEFAULT 'medium',
    semi_urban_suitability ENUM('high', 'medium', 'low') NOT NULL DEFAULT 'medium',
    rural_suitability ENUM('high', 'medium', 'low') NOT NULL DEFAULT 'medium',

    market_dependency_notes TEXT NULL,
    raw_material_dependency_notes TEXT NULL,
    first_income_timeline TEXT NULL,
    success_tips TEXT NULL,
    common_mistakes TEXT NULL,

    status ENUM('draft', 'published', 'archived') NOT NULL DEFAULT 'draft',
    featured_flag TINYINT(1) NOT NULL DEFAULT 0,

    created_by BIGINT UNSIGNED NULL,
    updated_by BIGINT UNSIGNED NULL,
    published_at DATETIME NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    PRIMARY KEY (id),
    UNIQUE KEY uq_opportunities_slug (slug),
    KEY idx_opportunities_category (category_id),
    KEY idx_opportunities_type (opportunity_type),
    KEY idx_opportunities_status (status),
    KEY idx_opportunities_featured (featured_flag),
    KEY idx_opportunities_home_based (home_based_suitability),
    KEY idx_opportunities_investment (investment_min, investment_max),
    CONSTRAINT fk_opportunities_category
        FOREIGN KEY (category_id) REFERENCES jobs_opportunity_categories (id)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_opportunities_created_by
        FOREIGN KEY (created_by) REFERENCES jobs_admin_users (id)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    CONSTRAINT fk_opportunities_updated_by
        FOREIGN KEY (updated_by) REFERENCES jobs_admin_users (id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
) ;

        ON UPDATE CASCADE,

        FOREIGN KEY (state_id) REFERENCES jobs_states (id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT fk_opportunity_location_rules_district
        FOREIGN KEY (district_id) REFERENCES jobs_districts (id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT chk_opportunity_location_rules_ref
        CHECK (
            state_id IS NOT NULL OR district_id IS NOT NULL
        )
) ;

        ON DELETE SET NULL
        ON UPDATE CASCADE,
    CONSTRAINT fk_user_feedback_opportunity
        FOREIGN KEY (opportunity_id) REFERENCES jobs_opportunities (id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
) ;
